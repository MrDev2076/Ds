Problem Statement: Bus Seat Fragmentation

Imagine you are building a bus booking system. A core challenge is to find a block of **N contiguous empty seats** for a group booking. As seats are booked and released, the bus layout can become "fragmented" with scattered empty seats, making it impossible to find a large enough block, even if the total number of empty seats is sufficient. This is known as **external fragmentation**.

  1. Initial State: The bus is represented as a linear array of seats, arranged row by row. Each seat has one of three states: Empty (ðŸŸ¢), Reserved (ðŸŸ¡), or Confirmed (ðŸ”´).
  2. Group Booking: A user requests to book a specific number of seats ($N$) together. The system must find the **first** available block of $N$ contiguous empty seats.
  3. The Fragmentation Challenge: After many bookings and cancellations (simulated by releasing seats), the bus may have enough total empty seats, but no large enough contiguous block to satisfy a new group booking. The system must detect this and notify the user.

Seat States:

  • Empty (ðŸŸ¢): Available for booking.
  • Reserved (ðŸŸ¡): Temporarily held for a pending booking. This state is not directly used in this simple visualizer but is a key concept in a real system. We will simply move from empty to booked.
  • Confirmed (ðŸ”´): Successfully booked and permanently unavailable.

import java.util.Scanner;
import java.util.Arrays;

public class BusBookingFragmentation {
    static int SEATS_PER_ROW = 4;
    static int TOTAL_SEATS = 12;
    static int[] busSeats = new int[TOTAL_SEATS]; // 0: Empty, 1: Reserved, 2: Booked

    public static void initializeBus() {
        Arrays.fill(busSeats, 0);
    }

    public static boolean bookSeats(int numSeats) {
        int emptyCount = 0;
        int startIndex = -1;

        // Iterate through the seats to find a contiguous block
        for (int i = 0; i < TOTAL_SEATS; i++) {
            if (busSeats[i] == 0) {
                if (emptyCount == 0) {
                    startIndex = i;
                }
                emptyCount++;
                if (emptyCount == numSeats) {
                    // Found a block, book the seats
                    for (int j = startIndex; j < startIndex + numSeats; j++) {
                        busSeats[j] = 2; // Mark as booked (2)
                    }
                    return true;
                }
            } else {
                emptyCount = 0; // Reset counter if a non-empty seat is found
            }
        }
        return false; // No contiguous block of sufficient size found
    }

    public static void printBusLayout() {
        System.out.println("\nCurrent Bus Layout (0:Empty, 2:Booked):");
        for (int i = 0; i < TOTAL_SEATS; i++) {
            System.out.print(busSeats[i] + " ");
            if ((i + 1) % SEATS_PER_ROW == 0) {
                System.out.println();
            }
        }
        System.out.println();
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        initializeBus();
        printBusLayout();

        while (true) {
            System.out.print("Enter number of seats to book (0 to exit): ");
            int numSeats = scanner.nextInt();
            if (numSeats == 0) {
                System.out.println("Exiting.");
                break;
            }
            if (bookSeats(numSeats)) {
                System.out.println("Successfully booked " + numSeats + " seats.");
            } else {
                System.out.println("Fragmentation Error: Cannot find " + numSeats + " contiguous seats.");
            }
            printBusLayout();
        }
        scanner.close();
    }
}
